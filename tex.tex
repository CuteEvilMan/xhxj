\section*{连续时间傅里叶变换 (CTFT)}

\subsection*{一、重要性质}
设 $x(t) \leftrightarrow X(j\omega)$, $y(t) \leftrightarrow Y(j\omega)$。
\begin{enumerate}
    \item \textbf{相乘 (时域)}: $x(t)y(t) \leftrightarrow \frac{1}{2\pi}\int_{-\infty}^{\infty}X(j\theta)Y(j(\omega-\theta))d\theta$
    \item \textbf{时域微分}: $\frac{d}{dt}x(t) \leftrightarrow j\omega X(j\omega)$
    \item \textbf{积分}: $\int_{-\infty}^{t}x(\tau)d\tau \leftrightarrow \frac{1}{j\omega}X(j\omega)+\pi X(0)\delta(\omega)$
    \item \textbf{频域微分}: $tx(t) \leftrightarrow j\frac{d}{d\omega}X(j\omega)$
    \item \textbf{实信号共轭对称}: $x(t)$ 实 $\Rightarrow X(j\omega) = X^*(-j\omega)$
    \\[2pt] $\text{Re}[X(j\omega)]$ 偶; $\text{Im}[X(j\omega)]$ 奇
    \item \textbf{实偶/奇信号}: 实偶 $\leftrightarrow$ 实偶; 实奇 $\leftrightarrow$ 纯虚奇
    \item \textbf{奇偶分解}: $x_{\text{e}}(t) \leftrightarrow \text{Re}[X(j\omega)]$; $x_{\text{o}}(t) \leftrightarrow j \text{Im}[X(j\omega)]$
    \item \textbf{帕斯瓦尔}: $\int_{-\infty}^{+\infty}|x(t)|^2 dt = \frac{1}{2\pi}\int_{-\infty}^{+\infty}|X(j\omega)|^2 d\omega$
\end{enumerate}

\subsection*{二、基本变换对}
\begin{enumerate}
    \item $e^{j\omega_0 t} \leftrightarrow 2\pi\delta(\omega-\omega_0)$
    \item $\cos(\omega_0 t) \leftrightarrow \pi[\delta(\omega-\omega_0)+\delta(\omega+\omega_0)]$
    \item $\sin(\omega_0 t) \leftrightarrow \frac{\pi}{j}[\delta(\omega-\omega_0)-\delta(\omega+\omega_0)]$
    \item $1 \leftrightarrow 2\pi\delta(\omega)$
    \item $\sum_{k} \delta(t-kT) \leftrightarrow \frac{2\pi}{T}\sum_{k}\delta(\omega-\frac{2\pi k}{T})$
    \item $\delta(t) \leftrightarrow 1$
    \item $\delta(t-t_0) \leftrightarrow e^{-j\omega t_0}$
    \item $u(t) \leftrightarrow \frac{1}{j\omega} + \pi\delta(\omega)$
    \item $e^{-at}u(t), \text{Re}\{a\}>0 \leftrightarrow \frac{1}{a+j\omega}$
    \item $t e^{-at}u(t), \text{Re}\{a\}>0 \leftrightarrow \frac{1}{(a+j\omega)^2}$
    \item $t^{n-1} e^{-at}u(t), \text{Re}\{a\}>0 \leftrightarrow \frac{(n-1)!}{(a+j\omega)^n}$
    \item \textbf{周期方波} $\leftrightarrow \frac{2\omega_0 T_1}{T} \sum_{k} \operatorname{sinc}\left(\frac{k\omega_0 T_1}{\pi}\right) \delta(\omega-k\omega_0)$
\end{enumerate}

\section*{离散时间傅里叶变换 (DTFT)}

\subsection*{一、重要性质 (表 5.3)}
设 $x[n] \leftrightarrow X(e^{j\omega})$, $y[n] \leftrightarrow Y(e^{j\omega})$。
\begin{enumerate}
    \item \textbf{时域扩展}: $x_{(k)}[n] \leftrightarrow X(e^{jk\omega})$ ($n$非$k$倍数为0)
    \item \textbf{卷积}: $x[n] * y[n] \leftrightarrow X(e^{j\omega})Y(e^{j\omega})$
    \item \textbf{相乘}: $x[n]y[n] \leftrightarrow \frac{1}{2\pi}\int_{2\pi}X(e^{j\theta})Y(e^{j(\omega-\theta)})d\theta$
    \item \textbf{时域差分}: $x[n] - x[n-1] \leftrightarrow (1 - e^{-j\omega})X(e^{j\omega})$
    \item \textbf{频域微分}: $n x[n] \leftrightarrow j \frac{d}{d\omega} X(e^{j\omega})$
    \item \textbf{实信号对称}: $x[n]$实 $\leftrightarrow X(e^{j\omega}) = X^*(e^{-j\omega})$
    \item \textbf{实偶/奇}: 实偶 $\leftrightarrow$ 实偶; 实奇 $\leftrightarrow$ 纯虚奇
    \item \textbf{奇偶分解}: $x_{\text{e}}[n] \leftrightarrow \text{Re}[X(e^{j\omega})]$; $x_{\text{o}}[n] \leftrightarrow j \text{Im}[X(e^{j\omega})]$
    \item \textbf{帕斯瓦尔}: $\sum |x[n]|^2 = \frac{1}{2\pi}\int_{2\pi}|X(e^{j\omega})|^2 d\omega$
\end{enumerate}

\subsection*{二、基本变换对 (表 5.2)}
\begin{enumerate}
    \item \textbf{DTFS}: $\sum_{k} \delta[n-kN] \leftrightarrow \frac{2\pi}{N}\sum_{r} \delta(\omega-\frac{2\pi r}{N})$
    \item $e^{j\omega_0 n} \leftrightarrow 2\pi\sum_{k} \delta(\omega-\omega_0 - 2\pi k)$
    \item $\cos(\omega_0 n) \leftrightarrow \pi\sum_{k} [\delta(\omega-\omega_0 - 2\pi k) + \delta(\omega+\omega_0 - 2\pi k)]$
    \item $\sin(\omega_0 n) \leftrightarrow \frac{\pi}{j}\sum_{k} [\delta(\omega-\omega_0 - 2\pi k) - \delta(\omega+\omega_0 - 2\pi k)]$
    \item $a^n u[n], |a|<1 \leftrightarrow \frac{1}{1 - a e^{-j\omega}}$
    \item \textbf{矩形脉冲}: $x[n]=1 (|n|\le N_1) \leftrightarrow \frac{\sin[\omega (N_1 + 1/2)]}{\sin(\omega/2)}$
    \item \textbf{低通}: $\text{Rect}_W(\omega) \leftrightarrow \frac{\sin(Wn)}{\pi n}$
    \item $\delta[n] \leftrightarrow 1$
    \item $u[n] \leftrightarrow \frac{1}{1 - e^{-j\omega}} + \pi\sum_{k}\delta(\omega-2\pi k)$
    \item $\delta[n-n_0] \leftrightarrow e^{-j\omega n_0}$
    \item $(n+1)a^n u[n], |a|<1 \leftrightarrow \frac{1}{(1-a e^{-j\omega})^2}$
    \item $\frac{(n+r-1)!}{n!(r-1)!} a^n u[n], |a|<1 \leftrightarrow \frac{1}{(1-a e^{-j\omega})^r}$
\end{enumerate}

\section*{主要变换定义公式}

\subsection*{一、傅里叶级数 (FS)}
\textbf{1. 连续 (CTFS):}
$$a_{k} = \frac{1}{\mathcal{T}} \int_{\mathcal{T}} x(t) e^{-j k \omega_0 t} dt, \quad x(t) = \sum_{k=-\infty}^{+\infty} a_k e^{j k \omega_0 t}$$

\textbf{2. 离散 (DTFS):}
$$a_k = \frac{1}{N} \sum_{n=\langle N \rangle} x[n] e^{-j k \frac{2\pi}{N} n}, \quad x[n] = \sum_{k=\langle N \rangle} a_k e^{j k \frac{2\pi}{N} n}$$

\subsection*{二、傅里叶变换 (FT)}
\textbf{3. 连续 (CTFT):}
$$X(j\omega) = \int_{-\infty}^{\infty} x(t) e^{-j\omega t} dt, \quad x(t) = \frac{1}{2\pi} \int_{-\infty}^{\infty} X(j\omega) e^{j\omega t} d\omega$$

\textbf{4. 离散 (DTFT):}
$$X(e^{j\omega}) = \sum_{n=-\infty}^{\infty} x[n] e^{-j\omega n}, \quad x[n] = \frac{1}{2\pi} \int_{2\pi} X(e^{j\omega}) e^{j\omega n} d\omega$$

\subsection*{三、复平面变换 (S域与Z域)}
\textbf{5. 连续拉普拉斯 (CTLT):}
$$X(s) = \int_{-\infty}^{+\infty} x(t) e^{-st} dt, \quad x(t) = \frac{1}{2\pi j} \int_{\sigma-j\infty}^{\sigma+j\infty} X(s) e^{st} ds$$

\textbf{6. Z 变换 (DTLT):}
$$X(z) = \sum_{n=-\infty}^{+\infty} x[n] z^{-n}, \quad x[n] = \frac{1}{2\pi j} \oint_C X(z) z^{n-1} dz$$

\textbf{7. 单边Z变换:}
$$X_+(z) = \sum_{n=0}^{+\infty} x[n] z^{-n}, \quad x[n] = \frac{1}{2\pi j} \oint_C X_+(z) z^{n-1} dz$$

\section*{拉普拉斯变换 (CTLT)}

\subsection*{一、重要性质 (表 9.1)}
\begin{enumerate}
    \item \textbf{线性}: $a X_1(s) + b X_2(s)$ (ROC至少 $R_1 \cap R_2$)
    \item \textbf{时移}: $x(t-t_0) \leftrightarrow e^{-st_0}X(s)$ (ROC不变)
    \item \textbf{$s$平移}: $e^{s_0 t}x(t) \leftrightarrow X(s-s_0)$ (ROC平移)
    \item \textbf{尺度}: $x(at) \leftrightarrow \frac{1}{|a|}X(\frac{s}{a})$ (ROC变为 $aR$)
    \item \textbf{共轭}: $x^*(t) \leftrightarrow X^*(s^*)$
    \item \textbf{卷积}: $x_1 * x_2 \leftrightarrow X_1(s) X_2(s)$ (ROC $\cap$)
    \item \textbf{时域微分}: $x'(t) \leftrightarrow s X(s) - x(0^-)$ (单边考虑初始值)
    \item \textbf{$s$域微分}: $-t x(t) \leftrightarrow \frac{d}{ds}X(s)$
    \item \textbf{积分}: $\int_{-\infty}^{t}x(\tau)d\tau \leftrightarrow \frac{1}{s}X(s)$ (ROC $\cap \text{Re}(s)>0$)
    \item \textbf{初/终值}: $x(0^{+})=\lim_{s\rightarrow\infty}sX(s)$; $\lim_{t\rightarrow\infty}x(t) = \lim_{s\rightarrow 0}sX(s)$
\end{enumerate}

\subsection*{二、基本变换对 (表 9.2)}
\textbf{注：}以下均为右边信号 (ROC通常为 $\text{Re}\{s\} > \text{pole}$)
\begin{enumerate}
    \item $\delta(t) \leftrightarrow 1$ (全平面)
    \item $u(t) \leftrightarrow \frac{1}{s}$
    \item $-u(-t) \leftrightarrow \frac{1}{s} \quad (\text{Re}\{s\}<0)$
    \item $e^{-at}u(t) \leftrightarrow \frac{1}{s+a}$
    \item $t e^{-at}u(t) \leftrightarrow \frac{1}{(s+a)^2}$
    \item $\frac{t^{n-1}}{(n-1)!} u(t) \leftrightarrow \frac{1}{s^n}$
    \item $\frac{t^{n-1}}{(n-1)!} e^{-at}u(t) \leftrightarrow \frac{1}{(s+a)^n}$
    \item $[\cos \omega_0 t]u(t) \leftrightarrow \frac{s}{s^2+\omega_0^2}$
    \item $[\sin \omega_0 t]u(t) \leftrightarrow \frac{\omega_0}{s^2+\omega_0^2}$
\end{enumerate}

\subsection*{三、收敛域 (ROC)}
\begin{itemize}
    \item \textbf{因果}: ROC为右侧半平面。
    \item \textbf{反因果}: ROC为左侧半平面。
    \item \textbf{稳定}: ROC包含虚轴 ($\text{Re}\{s\}=0$)。
\end{itemize}

\subsection*{四、单边拉普拉斯性质 (表 9.3)}
假设 $t<0$ 时 $x(t)=0$。
\begin{enumerate}
    \item \textbf{线性/平移/尺度/卷积}: 与双边类似。
    \item \textbf{时域微分}: $x'(t) \leftrightarrow s X(s) - x(0^-)$
    \item \textbf{$s$域微分}: $-t x(t) \leftrightarrow X'(s)$
    \item \textbf{初值}: $x(0^{+})=\lim_{s\rightarrow\infty}sX(s)$
    \item \textbf{终值}: $\lim_{t\rightarrow\infty}x(t) = \lim_{s\rightarrow 0}sX(s)$
\end{enumerate}

\section*{Z 变换 (DTLT)}

\subsection*{一、重要性质 (表 10.1)}
设 $x[n] \leftrightarrow X(z)$。
\begin{enumerate}
    \item \textbf{线性}: $a X_1(z) + b X_2(z)$
    \item \textbf{时移}: $x[n-n_0] \leftrightarrow z^{-n_0} X(z)$
    \item \textbf{尺度}: $a^n x[n] \leftrightarrow X(a^{-1}z)$ (ROC缩放)
    \item \textbf{反转}: $x[-n] \leftrightarrow X(z^{-1})$ (ROC倒数)
    \item \textbf{扩展}: $x_{(k)}[n] \leftrightarrow X(z^k)$
    \item \textbf{共轭}: $x^*[n] \leftrightarrow X^*(z^*)$
    \item \textbf{卷积}: $x_1 * x_2 \leftrightarrow X_1(z) X_2(z)$
    \item \textbf{一次差分}: $x[n] - x[n-1] \leftrightarrow (1 - z^{-1}) X(z)$
    \item \textbf{累加}: $\sum_{k=-\infty}^{n} x[k] \leftrightarrow \frac{1}{1-z^{-1}} X(z)$
    \item \textbf{$z$域微分}: $n x[n] \leftrightarrow -z \frac{d}{dz} X(z)$
    \item \textbf{初值}: 若因果, $x[0] = \lim_{z\rightarrow\infty} X(z)$
\end{enumerate}

\subsection*{二、单边 Z 变换性质 (表 10.3)}
假设 $n<0$ 时 $x[n]=0$。
\begin{enumerate}
    \item \textbf{右移}: $x[n-1] \leftrightarrow z^{-1}X(z) + x[-1]$
    \item \textbf{左移}: $x[n+1] \leftrightarrow zX(z) - zx[0]$
    \item \textbf{一次差分}: $x[n] - x[n-1] \leftrightarrow (1 - z^{-1}) X(z) - x[-1]$
\end{enumerate}

\subsection*{三、基本变换对 (表 10.2 整理)}
格式：信号 $\leftrightarrow$ 变换, ROC
\begin{enumerate}
    \item $\delta[n] \leftrightarrow 1$, All $z$
    \item $u[n] \leftrightarrow \frac{1}{1-z^{-1}}$, $|z|>1$
    \item $-u[-n-1] \leftrightarrow \frac{1}{1-z^{-1}}$, $|z|<1$
    \item $\delta[n-m] \leftrightarrow z^{-m}$, $z \neq 0,\infty$
    \item $a^n u[n] \leftrightarrow \frac{1}{1-a z^{-1}}$, $|z|>|a|$
    \item $-a^n u[-n-1] \leftrightarrow \frac{1}{1-a z^{-1}}$, $|z|<|a|$
    \item $n a^n u[n] \leftrightarrow \frac{a z^{-1}}{(1-a z^{-1})^2}$, $|z|>|a|$
    \item $-n a^n u[-n-1] \leftrightarrow \frac{a z^{-1}}{(1-a z^{-1})^2}$, $|z|<|a|$
    \item $[\cos \omega_0 n]u[n] \leftrightarrow \frac{1 - (\cos \omega_0) z^{-1}}{1 - (2 \cos \omega_0) z^{-1} + z^{-2}}$, $|z|>1$
    \item $[\sin \omega_0 n]u[n] \leftrightarrow \frac{(\sin \omega_0) z^{-1}}{1 - (2 \cos \omega_0) z^{-1} + z^{-2}}$, $|z|>1$
\end{enumerate}
